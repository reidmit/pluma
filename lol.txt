  . age : mut<int>

  | rename : string mut<self> -> done

: mut<renamable> string -> done
change-name-to-reid = \ thing ->
  thing | rename "reid"


# mutates its person argument
: mut<person> -> void
have-birthday = \ p ->
  p.age := p.age + 1


# returns a new person
: person -> person
one-year-older = \ p ->
  person (...p, age: p.age + 1)


# execution starts here!
: entrypoint
main = \
  me = mut (person (name: "reid", age: 28));
  me | have-birthday;

: int -> list<int>
get-primes = \ max ->
  let acc = [];

  range (0, max) \ curr ->
    mut found = false;

    for acc \ d ->
      if curr % d == 0 \
        found := true;

    if (not found) \
      acc | push curr;

    done;

  acc

: list<int> int -> bool
